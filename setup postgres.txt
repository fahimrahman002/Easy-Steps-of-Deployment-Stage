1. sudo apt install postgresql -y
2. sudo su postgres
3. psql -U  postgres -c "CREATE ROLE ubuntu;"
4. psql -U  postgres -c "ALTER ROLE ubuntu WITH LOGIN;"
5. psql -U  postgres -c "ALTER ROLE ubuntu CREATEDB;"
6. psql -U  postgres -c "ALTER ROLE ubuntu WITH PASSWORD '1234';"
7. sudo find / -name "postgresql.conf" or locate "postgresql.conf"
Then edit /etc/postgresql/12/main/postgresql.conf
and add 
listen_addresses = '*' 
8. locate pg_hba.conf
Add at the bottom-
host	all		all         0.0.0.0/0		md5
host	all		all		::/0		md5
9. sudo systemctl restart postgresql
